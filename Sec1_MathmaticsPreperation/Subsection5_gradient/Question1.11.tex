\begin{abstract}
  \[
    f(x, y, z) = e^x \cos(yz)
  \]
  $\grad f$の計算. 
\end{abstract}

\section*{計算}
定義どおりに計算するだけ. 
\begin{eqnarray*}
  \frac{\partial f}{\partial x} = e^x \cos(yz) \\
  \frac{\partial f}{\partial y} = -z e^x \sin(yz) \\
  \frac{\partial f}{\partial z} = -y e^x \sin(yz)
\end{eqnarray*}
